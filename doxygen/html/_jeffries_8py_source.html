34"""
35 Planimetric Procedure—The planimetric method involves an actual count of the number of grains within a known
36area. The number of grains per unit area, NA , is used to determine the ASTM grain size number, G. The precision of
37the method is a function of the number of grains counted. A precision of 60.25 grain size units can be attained with a
38reasonable amount of effort. Results are free of bias and repeatability and reproducibility are less than 60.5 grain size
39units. An accurate count does require marking off of the grains as they are counted.
40Standard Test Methods for Determining Average Grain Size
41 """

84def get_grain_size_related_parameters(N_inside, N_intercepted, M):
85 """
86 N_inside - Number of grains inside circunferene
87 N_intercepted - NUmber of grains intercepted by circunference
88
89 Returns:
90 Na - number of grains per mm2
91 A_mean - mean grain area
92 d - mean grain diamenter
93 G - ASTM grain size number
94 """
95
96 f = 0.0002*M # Jeffries multiplier
97 Na = f*(N_inside + 0.5*N_intercepted) # Number of grains per square millimetre
98 A_mean = 1/Na
99 d_mean = np.sqrt(A_mean)
100 G = 3.321928*np.log10(Na) - 2.954
101
102 return Na, A_mean, d_mean, G

105def run_jeffries_analysis(image_analyzer, window, image_prop):
106 """Jeffries procedure is applied N times to the same image for different
107 patterns. For each one, a G number is calculated and then an histogram built.
108 N - number of tests"""
109
110 "The ASTM article E-112 suggests a circunfence of 5000mm2 which gives R=39,894228mm"
111 ret = image_analyzer.find_grains()
112 if ret==-1:
113 return
114
115 width, height = image_analyzer.get_image_dimensions()
116
117 # number of tests field
118 n = 60
119 n_bins = int(n/5)
120 w_offset = int(image_prop.width*0)
121 h_offset = int(image_prop.height*0)
122
123 R = get_radius(image_prop)
124
125 "Define the region where the center of circunference can be placed."
126 "W: {(x,y): w_offset+R <= x <= width - R - w_offset, h_offset+R <= y <= height - R - h_offset"
127 w_lower = w_offset + R
128 w_upper = image_prop.width - (R + w_offset)
129 h_lower = h_offset + R
130 h_upper = image_prop.height - (R + h_offset)
131
132 # define and initialize circunference
133 circunference = [R, (0,0)]
134
135 G_arr = np.zeros(n)
136 for i in range(n):
137 x = random.randint(w_lower, w_upper)
138 y = random.randint(h_lower, h_upper)
139 circunference[1] = (x,y)
140
141 in_list, inter_list = Jeffries(image_analyzer, circunference)
142 N_inside = len(in_list)
143 N_inter = len(inter_list)
144 # print("grains %d (inside) %d (intercepted)" % (N_inside, N_inter ))
145 Na, A_mean, d_mean, G = get_grain_size_related_parameters(N_inside, N_inter, image_prop)
146 # print(G)
147 G_arr[i] = G
148
149 geo.draw_circle((x,y), R, window)
150 in_list.clear()
151 inter_list.clear()
152
153 return G_arr, n_bins
