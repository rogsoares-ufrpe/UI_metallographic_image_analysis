653from time import time, localtime, strftime

655class ToolTip(tk.Toplevel):
656 """
657 Provides a ToolTip widget for Tkinter.
658 To apply a ToolTip to any Tkinter widget, simply pass the widget to the
659 ToolTip constructor
660 """
661 def __init__(self, wdgt, tooltip_font, msg=None, msgFunc=None,
662 delay=1, follow=True):
663 """
664 Initialize the ToolTip
665
666 Arguments:
667 wdgt: The widget this ToolTip is assigned to
668 tooltip_font: Font to be used
669 msg: A static string message assigned to the ToolTip
670 msgFunc: A function that retrieves a string to use as the ToolTip text
671 delay: The delay in seconds before the ToolTip appears(may be float)
672 follow: If True, the ToolTip follows motion, otherwise hides
673 """
674 self.wdgt = wdgt
675 # The parent of the ToolTip is the parent of the ToolTips widget
676 self.parent = self.wdgt.master
677 # Initalise the Toplevel
678 tk.Toplevel.__init__(self, self.parent, bg='black', padx=1, pady=1)
679 # Hide initially
680 self.withdraw()
681 # The ToolTip Toplevel should have no frame or title bar
682 self.overrideredirect(True)
683
684 # The msgVar will contain the text displayed by the ToolTip
685 self.msgVar = tk.StringVar()
686 if msg is None:
687 self.msgVar.set('No message provided')
688 else:
689 self.msgVar.set(msg)
690 self.msgFunc = msgFunc
691 self.delay = delay
692 self.follow = follow
693 self.visible = 0
694 self.lastMotion = 0
695 # The text of the ToolTip is displayed in a Message widget
696 tk.Message(self, textvariable=self.msgVar, bg='#FFFFDD',
697 font=tooltip_font,
698 aspect=1000).grid()
699
700 # Add bindings to the widget. This will NOT override
701 # bindings that the widget already has
702 self.wdgt.bind('<Enter>', self.spawnspawn, '+')
703 self.wdgt.bind('<Leave>', self.hidehide, '+')
704 self.wdgt.bind('<Motion>', self.movemove, '+')
705
706 def spawn(self, event=None):
707 """
708 Spawn the ToolTip. This simply makes the ToolTip eligible for display.
709 Usually this is caused by entering the widget
710
711 Arguments:
712 event: The event that called this funciton
713 """
714 self.visible = 1
715 # The after function takes a time argument in miliseconds
716 self.after(int(self.delay * 1000), self.showshow)
717
718 def show(self):
719 """
720 Displays the ToolTip if the time delay has been long enough
721 """
722 if self.visible == 1 and time() - self.lastMotion > self.delay:
723 self.visible = 2
724 if self.visible == 2:
725 self.deiconify()
726
727 def move(self, event):
728 """
729 Processes motion within the widget.
730 Arguments:
731 event: The event that called this function
732 """
733 self.lastMotion = time()
734 # If the follow flag is not set, motion within the
735 # widget will make the ToolTip disappear
736 #
737 if self.follow is False:
738 self.withdraw()
739 self.visible = 1
740
741 # Offset the ToolTip 10x10 pixes southwest of the pointer
742 self.geometry('+%i+%i' % (event.x_root+20, event.y_root-10))
743 try:
744 # Try to call the message function. Will not change
745 # the message if the message function is None or
746 # the message function fails
747 self.msgVar.set(self.msgFunc())
748 except:
749 pass
750 self.after(int(self.delay * 1000), self.showshow)
751
752 def hide(self, event=None):
753 """
754 Hides the ToolTip. Usually this is caused by leaving the widget
755 Arguments:
756 event: The event that called this function
757 """
758 self.visible = 0
759 self.withdraw()
