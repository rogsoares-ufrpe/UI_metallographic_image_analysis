96def line_circle_intercept(pt1, pt2, center, radius):
97 isIntercept = False
98 x0 = center[0]
99 y0 = center[1]
100 R = radius
101
102 x1 = pt1[0]
103 y1 = pt1[1]
104 x2 = pt2[0]
105 y2 = pt2[1]
106
107 a = x1 - x2
108 b = x2 - x0
109 c = y1 - y2
110 d = y2 - y0
111
112 A = a**2 + c**2
113 B = 2*(a*b+c*d)
114 C = b**2 + d**2 - R**2
115
116 delta = B**2 - 4.*A*C
117 if delta > 0:
118 t1 = (-B + np.sqrt(delta))/(2.*A)
119 t2 = (-B - np.sqrt(delta))/(2.*A)
120 if 0 <= t1 <= 1 or 0 <= t2 <= 1:
121 isIntercept = True
122
123 return isIntercept
