741class ToolTip(tk.Toplevel):
742 """
743 Provides a ToolTip widget for Tkinter.
744 To apply a ToolTip to any Tkinter widget, simply pass the widget to the
745 ToolTip constructor
746 """
747 def __init__(self, wdgt, tooltip_font, msg=None, msgFunc=None,
748 delay=1, follow=True):
749 """
750 Initialize the ToolTip
751
752 Arguments:
753 wdgt: The widget this ToolTip is assigned to
754 tooltip_font: Font to be used
755 msg: A static string message assigned to the ToolTip
756 msgFunc: A function that retrieves a string to use as the ToolTip text
757 delay: The delay in seconds before the ToolTip appears(may be float)
758 follow: If True, the ToolTip follows motion, otherwise hides
759 """
760 self.wdgt = wdgt
761 # The parent of the ToolTip is the parent of the ToolTips widget
762 self.parent = self.wdgt.master
763 # Initalise the Toplevel
764 tk.Toplevel.__init__(self, self.parent, bg='black', padx=1, pady=1)
765 # Hide initially
766 self.withdraw()
767 # The ToolTip Toplevel should have no frame or title bar
768 self.overrideredirect(True)
769
770 # The msgVar will contain the text displayed by the ToolTip
771 self.msgVar = tk.StringVar()
772 if msg is None:
773 self.msgVar.set('No message provided')
774 else:
775 self.msgVar.set(msg)
776 self.msgFunc = msgFunc
777 self.delay = delay
778 self.follow = follow
779 self.visible = 0
780 self.lastMotion = 0
781 # The text of the ToolTip is displayed in a Message widget
782 tk.Message(self, textvariable=self.msgVar, bg='#FFFFDD',
783 font=tooltip_font,
784 aspect=1000).grid()
785
786 # Add bindings to the widget. This will NOT override
787 # bindings that the widget already has
788 self.wdgt.bind('<Enter>', self.spawnspawn, '+')
789 self.wdgt.bind('<Leave>', self.hidehide, '+')
790 self.wdgt.bind('<Motion>', self.movemove, '+')
791
792 def spawn(self, event=None):
793 """
794 Spawn the ToolTip. This simply makes the ToolTip eligible for display.
795 Usually this is caused by entering the widget
796
797 Arguments:
798 event: The event that called this funciton
799 """
800 self.visible = 1
801 # The after function takes a time argument in miliseconds
802 self.after(int(self.delay * 1000), self.showshow)
803
804 def show(self):
805 """
806 Displays the ToolTip if the time delay has been long enough
807 """
808 if self.visible == 1 and time() - self.lastMotion > self.delay:
809 self.visible = 2
810 if self.visible == 2:
811 self.deiconify()
812
813 def move(self, event):
814 """
815 Processes motion within the widget.
816 Arguments:
817 event: The event that called this function
818 """
819 self.lastMotion = time()
820 # If the follow flag is not set, motion within the
821 # widget will make the ToolTip disappear
822 #
823 if self.follow is False:
824 self.withdraw()
825 self.visible = 1
826
827 # Offset the ToolTip 10x10 pixes southwest of the pointer
828 self.geometry('+%i+%i' % (event.x_root+20, event.y_root-10))
829 try:
830 # Try to call the message function. Will not change
831 # the message if the message function is None or
832 # the message function fails
833 self.msgVar.set(self.msgFunc())
834 except:
835 pass
836 self.after(int(self.delay * 1000), self.showshow)
837
838 def hide(self, event=None):
839 """
840 Hides the ToolTip. Usually this is caused by leaving the widget
841 Arguments:
842 event: The event that called this function
843 """
844 self.visible = 0
845 self.withdraw()
