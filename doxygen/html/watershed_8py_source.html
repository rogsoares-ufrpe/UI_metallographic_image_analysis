36# Create a kernel that we will use to sharpen our image

38kernel = np.array([[1, 1, 1], [1, -8, 1], [1, 1, 1]], dtype=np.float32)

40# do the laplacian filtering as it is
41# well, we need to convert everything in something more deeper then CV_8U
42# because the kernel has some negative values,
43# and we can expect in general to have a Laplacian image with negative values
44# BUT a 8bits unsigned int (the one we are working with) can contain values from 0 to 255
45# so the possible negative number will be truncated
46imgLaplacian = cv.filter2D(src, cv.CV_32F, kernel)
47sharp = np.float32(src)
48imgResult = sharp - imgLaplacian

51imgResult = np.clip(imgResult, 0, 255)
