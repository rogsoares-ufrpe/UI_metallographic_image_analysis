124class ToolTip(tk.Toplevel):
125 """
126 Provides a ToolTip widget for Tkinter.
127 To apply a ToolTip to any Tkinter widget, simply pass the widget to the
128 ToolTip constructor
129 """
130 def __init__(self, wdgt, tooltip_font, msg=None, msgFunc=None,
131 delay=1, follow=True):
132 """
133 Initialize the ToolTip
134
135 Arguments:
136 wdgt: The widget this ToolTip is assigned to
137 tooltip_font: Font to be used
138 msg: A static string message assigned to the ToolTip
139 msgFunc: A function that retrieves a string to use as the ToolTip text
140 delay: The delay in seconds before the ToolTip appears(may be float)
141 follow: If True, the ToolTip follows motion, otherwise hides
142 """
143 self.wdgt = wdgt
144 # The parent of the ToolTip is the parent of the ToolTips widget
145 self.parent = self.wdgt.master
146 # Initalise the Toplevel
147 tk.Toplevel.__init__(self, self.parent, bg='black', padx=1, pady=1)
148 # Hide initially
149 self.withdraw()
150 # The ToolTip Toplevel should have no frame or title bar
151 self.overrideredirect(True)
152
153 # The msgVar will contain the text displayed by the ToolTip
154 self.msgVar = tk.StringVar()
155 if msg is None:
156 self.msgVar.set('No message provided')
157 else:
158 self.msgVar.set(msg)
159 self.msgFunc = msgFunc
160 self.delay = delay
161 self.follow = follow
162 self.visible = 0
163 self.lastMotion = 0
164 # The text of the ToolTip is displayed in a Message widget
165 tk.Message(self, textvariable=self.msgVar, bg='#FFFFDD',
166 font=tooltip_font,
167 aspect=1000).grid()
168
169 # Add bindings to the widget. This will NOT override
170 # bindings that the widget already has
171 self.wdgt.bind('<Enter>', self.spawnspawn, '+')
172 self.wdgt.bind('<Leave>', self.hidehide, '+')
173 self.wdgt.bind('<Motion>', self.movemove, '+')

175 def spawn(self, event=None):
176 """
177 Spawn the ToolTip. This simply makes the ToolTip eligible for display.
178 Usually this is caused by entering the widget
179
180 Arguments:
181 event: The event that called this funciton
182 """
183 self.visible = 1
184 # The after function takes a time argument in miliseconds
185 self.after(int(self.delay * 1000), self.showshow)
186
187 def show(self):
188 """
189 Displays the ToolTip if the time delay has been long enough
190 """
191 if self.visible == 1 and time() - self.lastMotion > self.delay:
192 self.visible = 2
193 if self.visible == 2:
194 self.deiconify()
195
196 def move(self, event):
197 """
198 Processes motion within the widget.
199 Arguments:
200 event: The event that called this function
201 """
202 self.lastMotion = time()
203 # If the follow flag is not set, motion within the
204 # widget will make the ToolTip disappear
205 #
206 if self.follow is False:
207 self.withdraw()
208 self.visible = 1
209
210 # Offset the ToolTip 10x10 pixes southwest of the pointer
211 self.geometry('+%i+%i' % (event.x_root+20, event.y_root-10))
212 try:
213 # Try to call the message function. Will not change
214 # the message if the message function is None or
215 # the message function fails
216 self.msgVar.set(self.msgFunc())
217 except:
218 pass
219 self.after(int(self.delay * 1000), self.showshow)
220
221 def hide(self, event=None):
222 """
223 Hides the ToolTip. Usually this is caused by leaving the widget
224 Arguments:
225 event: The event that called this function
226 """
227 self.visible = 0
228 self.withdraw()
