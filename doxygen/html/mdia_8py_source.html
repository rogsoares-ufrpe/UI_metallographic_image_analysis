131 """Perform Canny algorithm to find edges segmentation.
132 Gray image scale is created during loading image process """

154 # Edge segmentation
155 edges = cv2.Canny(self.image_gray,
156 threshold1=self.threshold_one,
157 threshold2=self.threshold_two,
158 L2gradient=self.l2grad)
159
160 # removing undersireble edges
161 edges = cv2.morphologyEx(edges, cv2.MORPH_CLOSE, None)
162 # edges = cv2.morphologyEx(edges, cv2.MORPH_CLOSE, None)
163 edges = cv2.ximgproc.thinning(edges, thinningType=THINNING_ZHANGSUEN)
164 # edges = cv2.ximgproc.thinning(edges, thinningType=THINNING_GUOHALL)
165
166 # realiza a poda dos contornos
167 edges = np.uint8(edges > 0)
168 count = 0
169 while count != np.sum(edges):
170 # non-zero pixel count
171 count = np.sum(edges)
172 # examine 3x3 neighborhood of each pixel
173 filt = cv2.boxFilter(edges, -1, (3, 3), normalize=False)
174 # if the center pixel of 3x3 neighborhood is zero, we are not interested in it
175 edges = edges * filt
176 # now we have pixels where the center pixel of 3x3 neighborhood is non-zero
177 # if a pixels' 8-connectivity is less than 2 we can remove it
178 # threshold is 3 here because the boxfilter also counted the center pixel
179 edges[edges < 3] = 0
180 # set max pixel value to 1
181 edges[edges > 0] = 1
182
183 num_rows = len(edges)
184 num_cols = len(edges[0])
185
186 for i in range(num_rows):
187 for j in range(num_cols):
188 if edges[i][j] == 1:
189 edges[i][j] = 255
190
191 # print('find contours')
192 contours, hierarchy = cv2.findContours(edges, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)
193
194 length = len(contours)
195 minarea = 5
196
197 self.grains_contours.clear()
198 self.grains_area.clear()
199
200 max_area = -1
201 pos = -1
202 counter = 0
203 for k in range(length):
204 area = cv2.contourArea(contours[k])
205 if area > minarea:
206 self.grains_contours.append(contours[k])
207 self.grains_area.append(area)
208 counter += 1
209
210 if area > max_area:
211 max_area = area
212 pos = counter
213
